Hint: As instances of LettersAndNumber are pairs of (string, number), use the method doubleValue from class Number to get numerical value assigned to a string.

public class Pair<T, U> {
	private T first;
	private U second;
	
	public Pair(T first, U second) {
		this.first = first;
		this.second = second;
	}

	public T getFirst() {
		return first;
	}

	public void setFirst(T first) {
		this.first = first;
	}

	public U getSecond() {
		return second;
	}

	public void setSecond(U second) {
		this.second = second;
	}
}
